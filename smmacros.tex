% ------------------ \MyQ quote -----------------
% #1: lang
% #2: label
% #3: text
\NewDocumentCommand{\MyQ}{m m +m}
{%
  % check for label
  \ifstrempty{#2}{\relax}{\smlabel{#2}}%
  % check for language
  {%
    \ifstrempty{#1}%
    {\begin{otherlanguage*}{spanish}}%
      % else
      {\begin{otherlanguage*}{#1}}%
        \ShL% debug
        % if spanish
        \ifstrequal{#1}{spanish}%
        % then
        {\enquote{#3}}%
        % else
        {%
          \ifstrequal{#1}{greek}%
          {#3}%
          {\textit{#3}}%
        }%
      \end{otherlanguage*}%
    }%
}


% ------------------ \MyC citation -----------------
% #1: language
% #2: label
% #3: original language text
% #4: translation text
% #5: reference
\NewDocumentCommand{\MyC}{m m +m +m m}
{%
  {%
    % check for label
    \ifstrempty{#2}{\relax}{\smlabel{#2}}%
      % select language of original quote
      \begin{otherlanguage*}{#1}%
        % #3: ---- original quote ----
        \ShL% debug
        % if spanish:
        \ifstrequal{#1}{spanish}%
          % then:
          {\enquote{#3}}%
          % else, if greek:
          {\ifstrequal{#1}{greek}%
              {\textup{{#3}}}%
              % else, all others 
              {\textit{#3}}%
            }%
          \end{otherlanguage*}% close either \begin{otherlanguage}
          % #4: trans.
          \ifstrempty{#4}{\relax}%
            {%
              {%
                \begin{otherlanguage*}{spanish}%
                  \ShL% debug
                  \space\enquote{#4}%
                \end{otherlanguage*}%
              }%
            }%
            % #5: ref
            \ifstrempty{#5}{\relax}{\space#5}%
            }%
          }
